\newcommand*{\roads}[2]{
    % define boolean for first iteration
    \def\lastpos{}
    \foreach \r in {#2}{
        % if \lastpos is not empty
        \ifx\lastpos\empty
        \else
            % draw road
            \draw[road=#1] \lastpos -- \r;
        \fi
        % set last position
        \xdef\lastpos{\r}
    }
}
